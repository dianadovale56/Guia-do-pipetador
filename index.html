// Adicione este código ao seu JavaScript existente document.addEventListener('DOMContentLoaded', function() { // ... (código existente) // Funcionalidade da Checklist const checklistCheckboxes = document.querySelectorAll('.checklist-checkbox'); const progressBar = document.getElementById('progressBar'); const progressText = document.getElementById('progressText'); const resetButton = document.getElementById('resetButton'); function updateChecklistProgress() { const total = checklistCheckboxes.length; const checked = Array.from(checklistCheckboxes).filter(cb => cb.checked).length; const percent = Math.round((checked / total) * 100); progressBar.style.width = `${percent}%`; progressText.textContent = `${percent}% completo`; // Salva no localStorage checklistCheckboxes.forEach((checkbox, index) => { localStorage.setItem(`checklist_${index}`, checkbox.checked); }); } // Carrega estado salvo checklistCheckboxes.forEach((checkbox, index) => { const savedState = localStorage.getItem(`checklist_${index}`); if (savedState === 'true') { checkbox.checked = true; } checkbox.addEventListener('change', updateChecklistProgress); }); // Botão reset resetButton.addEventListener('click', function() { checklistCheckboxes.forEach((checkbox, index) => { checkbox.checked = false; localStorage.removeItem(`checklist_${index}`); }); updateChecklistProgress(); }); // Atualiza inicialmente updateChecklistProgress(); });
